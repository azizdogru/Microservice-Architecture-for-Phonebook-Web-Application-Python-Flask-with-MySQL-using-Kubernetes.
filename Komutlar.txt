5. Mysql deploy ve service oluşturduk. 

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: mysql-deployment
  labels:
    app: mysql-db
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mysql-db
  template:
    metadata:
      labels:
        app: mysql-db #######BURASI İLE 
    spec:
      containers:
      - name: mysql
        image: mysql:5.7
        ports:
        - containerPort: 3306
        env:
          - name: MYSQL_ROOT_PASSWORD
            value: "C123c"
          - name: MYSQL_DATABASE
            value: "phonebook"
          - name: MYSQL_USER
            value: "clarus"
          - name: MYSQL_PASSWORD
            value: "sifre123"

---
apiVersion: v1
kind: Service
metadata:
  name: mysql-service
spec:
  selector:
    app: mysql-db #### BURASI AYNI OLMALI
  ports:
    - protocol: TCP
      port: 3306
      targetPort: 3306
